35 static size_t DownloadHandler(char* contents, size_t size, size_t nmemb, void* userp) {
36  assert(userp);
37  auto ds = static_cast<DownloadMetaStruct*>(userp);
38  uint64_t downloaded = size * nmemb;
39  uint64_t expected = ds->target.length();
40  if ((ds->downloaded_length + downloaded) > expected) {
41  return downloaded + 1; // curl will abort if return unexpected size;
42  }
43 
44  // incomplete writes will stop the download (written_size != nmemb*size)
45  size_t written_size = ds->fhandle->wfeed(reinterpret_cast<uint8_t*>(contents), downloaded);
46  ds->hasher().update(reinterpret_cast<const unsigned char*>(contents), written_size);
47 
48  ds->downloaded_length += downloaded;
49  return written_size;
50 }

81 bool PackageManagerInterface::fetchTarget(const Uptane::Target& target, Uptane::Fetcher& fetcher,
82  const KeyManager& keys, FetcherProgressCb progress_cb,
83  const api::FlowControlToken* token) {
84  (void)keys;
85  bool result = false;
86  try {
87  if (target.hashes().empty()) {
88  throw Uptane::Exception("image", "No hash defined for the target");
89  }
90  TargetStatus exists = PackageManagerInterface::verifyTarget(target);
91  if (exists == TargetStatus::kGood) {
92  LOG_INFO << "Image already downloaded; skipping download";
93  return true;
94  }
95  if (target.length() == 0) {
96  LOG_WARNING << "Skipping download of target with length 0";
97  return true;
98  }
99  std::unique_ptr<DownloadMetaStruct> ds = std_::make_unique<DownloadMetaStruct>(target, progress_cb, token);
100  if (exists == TargetStatus::kIncomplete) {
101  auto target_check = storage_->checkTargetFile(target);
102  ds->downloaded_length = target_check->first;
103  auto target_handle = storage_->openTargetFile(target);
104  ::restoreHasherState(ds->hasher(), target_handle.get());
105  target_handle->rclose();
106  ds->fhandle = target_handle->toWriteHandle();
107  } else {
108  // If the target was found, but is oversized or the hash doesn't match,
109  // just start over.
110  ds->fhandle = storage_->allocateTargetFile(false, target);
111  }
112 
113  std::string target_url = target.uri();
114  if (target_url.empty()) {
115  target_url = fetcher.getRepoServer() + "/targets/" + Utils::urlEncode(target.filename());
116  }
117 
118  HttpResponse response;
119  for (;;) {
120  response = http_->download(target_url, DownloadHandler, ProgressHandler, ds.get(),
121  static_cast<curl_off_t>(ds->downloaded_length));
122 
123  if (response.curl_code == CURLE_RANGE_ERROR) {
124  LOG_WARNING << "The image server doesn't support byte range requests,"
125  " try to download the image from the beginning: "
126  << target_url;
127  ds = std_::make_unique<DownloadMetaStruct>(target, progress_cb, token);
128  ds->fhandle = storage_->allocateTargetFile(false, target);
129  continue;
130  }
131 
132  if (!response.wasInterrupted()) {
133  break;
134  }
135  ds->fhandle.reset();
136  // sleep if paused or abort the download
137  if (!token->canContinue()) {
138  throw Uptane::Exception("image", "Download of a target was aborted");
139  }
140  ds->fhandle = storage_->openTargetFile(target)->toWriteHandle();
141  }
142  LOG_TRACE << "Download status: " << response.getStatusStr() << std::endl;
143  if (!response.isOk()) {
144  if (response.curl_code == CURLE_WRITE_ERROR) {
145  throw Uptane::OversizedTarget(target.filename());
146  }
147  throw Uptane::Exception("image", "Could not download file, error: " + response.error_message);
148  }
149  if (!target.MatchHash(Uptane::Hash(ds->hash_type, ds->hasher().getHexDigest()))) {
150  ds->fhandle->wabort();
151  throw Uptane::TargetHashMismatch(target.filename());
152  }
153  ds->fhandle->wcommit();
154  result = true;
155  } catch (const Uptane::Exception& e) {
156  LOG_WARNING << "Error while downloading a target: " << e.what();
157  }
158  return result;
159 }

161 TargetStatus PackageManagerInterface::verifyTarget(const Uptane::Target& target) const {
162  auto target_exists = storage_->checkTargetFile(target);
163  if (!target_exists) {
164  return TargetStatus::kNotFound;
165  } else if (target_exists->first < target.length()) {
166  return TargetStatus::kIncomplete;
167  } else if (target_exists->first > target.length()) {
168  return TargetStatus::kOversized;
169  }
170 
171  // Even if the file exists and the length matches, recheck the hash.
172  DownloadMetaStruct ds(target, nullptr, nullptr);
173  ds.downloaded_length = target_exists->first;
174  auto target_handle = storage_->openTargetFile(target);
175  ::restoreHasherState(ds.hasher(), target_handle.get());
176  target_handle->rclose();
177  if (!target.MatchHash(Uptane::Hash(ds.hash_type, ds.hasher().getHexDigest()))) {
178  LOG_ERROR << "Target exists with expected length, but hash does not match metadata! " << target;
179  return TargetStatus::kHashMismatch;
180  }
181 
182  return TargetStatus::kGood;
