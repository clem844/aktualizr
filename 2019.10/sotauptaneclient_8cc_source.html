79  if (!package_manager_->rebootDetected()) {
80  // nothing to do
81  return;
82  }
83 
84  LOG_INFO << "Device has been rebooted after an update";
85 
86  std::vector<Uptane::Target> updates;
87  unsigned int ecus_count = 0;
88  if (uptaneOfflineIteration(&updates, &ecus_count)) {
89  const Uptane::EcuSerial &ecu_serial = uptane_manifest.getPrimaryEcuSerial();
90 
91  std::vector<Uptane::Target> installed_versions;
92  boost::optional<Uptane::Target> pending_target;
93  storage->loadInstalledVersions(ecu_serial.ToString(), nullptr, &pending_target);
94 
95  if (!!pending_target) {
96  const std::string correlation_id = pending_target->correlation_id();
97 
98  data::InstallationResult install_res = package_manager_->finalizeInstall(*pending_target);
99  storage->saveEcuInstallationResult(ecu_serial, install_res);
100  if (install_res.success) {
101  storage->saveInstalledVersion(ecu_serial.ToString(), *pending_target, InstalledVersionUpdateMode::kCurrent);
102  report_queue->enqueue(std_::make_unique<EcuInstallationCompletedReport>(ecu_serial, correlation_id, true));
103  } else {
104  // finalize failed
105  // unset pending flag so that the rest of the uptane process can
106  // go forward again
107  storage->saveInstalledVersion(ecu_serial.ToString(), *pending_target, InstalledVersionUpdateMode::kNone);
108  report_queue->enqueue(std_::make_unique<EcuInstallationCompletedReport>(ecu_serial, correlation_id, false));
109  director_repo.dropTargets(*storage); // fix for OTA-2587, listen to backend again after end of install
110  }
111 
112  computeDeviceInstallationResult(nullptr, correlation_id);
113  putManifestSimple();
114  } else {
115  // nothing found on primary
116  LOG_ERROR << "Expected reboot after update on primary but no update found";
117  }
118  } else {
119  LOG_ERROR << "Invalid Uptane metadata in storage.";
120  }
121

126  data::InstallationResult result;
127  Uptane::EcuSerial ecu_serial = uptane_manifest.getPrimaryEcuSerial();
128 
129  // This is to recover more gracefully if the install process was interrupted
130  // but ends up booting the new version anyway (e.g: ostree finished
131  // deploying but the device restarted before the final saveInstalledVersion
132  // was called).
133  // By storing the version in the table (as uninstalled), we can still pick

141  // simple case: update already completed

305 void SotaUptaneClient::computeDeviceInstallationResult(data::InstallationResult *result,
306  const std::string &correlation_id) {
307  data::InstallationResult device_installation_result =
308  data::InstallationResult(data::ResultCode::Numeric::kOk, "Device has been successfully installed");
309  std::string raw_installation_report = "Installation succesful";
310 
311  do {
312  std::vector<std::pair<Uptane::EcuSerial, data::InstallationResult>> ecu_results;
313 
314  if (!storage->loadEcuInstallationResults(&ecu_results)) {
315  // failed to load ECUs' installation result
316  device_installation_result = data::InstallationResult(data::ResultCode::Numeric::kInternalError,
317  "Unable to get installation results from ecus");
318  raw_installation_report = "Failed to load ECUs' installation result";
319 
320  break;
321  }
322 
323  std::string result_code_err_str;
324 
325  for (const auto &r : ecu_results) {
326  auto ecu_serial = r.first;
327  auto installation_res = r.second;
328 
329  if (hw_ids.find(ecu_serial) == hw_ids.end()) {
330  // couldn't find any ECU with the given serial/ID
331  device_installation_result = data::InstallationResult(data::ResultCode::Numeric::kInternalError,
332  "Unable to get installation results from ecus");
333 
334  raw_installation_report = "Couldn't find any ECU with the given serial: " + ecu_serial.ToString();
335 
336  break;
337  }
338 
339  if (installation_res.needCompletion()) {
340  // one of the ECUs needs completion, aka an installation finalization
341  device_installation_result =
342  data::InstallationResult(data::ResultCode::Numeric::kNeedCompletion,
343  "ECU needs completion/finalization to be installed: " + ecu_serial.ToString());
344  raw_installation_report = "ECU needs completion/finalization to be installed: " + ecu_serial.ToString();
345 
346  break;
347  }
348 
349  // format:
350  // ecu1_hwid:failure1|ecu2_hwid:failure2
351  if (!installation_res.isSuccess()) {
352  std::string ecu_code_str = hw_ids.at(ecu_serial).ToString() + ":" + installation_res.result_code.toString();
353  result_code_err_str += (result_code_err_str != "" ? "|" : "") + ecu_code_str;
354  }
355  }
356 
357  if (!result_code_err_str.empty()) {
358  // installation on at least one of the ECUs has failed
359  device_installation_result =
360  data::InstallationResult(data::ResultCode(data::ResultCode::Numeric::kInstallFailed, result_code_err_str),
361  "Installation failed on at least one of ECUs");
362  raw_installation_report = "Installation failed on at least one of ECUs";
363 
364  break;
365  }
366 
367  } while (false);
368 
369  if (result != nullptr) {
370  *result = device_installation_result;
371  }
372 
373  // TODO: think of exception handling, the SQLite related code can throw exceptions
374  storage->storeDeviceInstallationResult(device_installation_result, raw_installation_report, correlation_id);
375 }

377 bool SotaUptaneClient::getNewTargets(std::vector<Uptane::Target> *new_targets, unsigned int *ecus_count) {
378  std::vector<Uptane::Target> targets = director_repo.getTargets();
379  Uptane::EcuSerial primary_ecu_serial = uptane_manifest.getPrimaryEcuSerial();
380  if (ecus_count != nullptr) {
381  *ecus_count = 0;
382  }
383  for (const Uptane::Target &target : targets) {
384  bool is_new = false;
385  for (const auto &ecu : target.ecus()) {
386  Uptane::EcuSerial ecu_serial = ecu.first;
387  Uptane::HardwareIdentifier hw_id = ecu.second;
388 
389  auto hwid_it = hw_ids.find(ecu_serial);
390  if (hwid_it == hw_ids.end()) {
391  LOG_ERROR << "Unknown ECU ID in director targets metadata: " << ecu_serial.ToString();
392  last_exception = Uptane::BadEcuId(target.filename());
393  return false;
394  }
395 
396  if (hwid_it->second != hw_id) {
397  LOG_ERROR << "Wrong hardware identifier for ECU " << ecu_serial.ToString();
398  last_exception = Uptane::BadHardwareId(target.filename());
399  return false;
400  }
401 
402  boost::optional<Uptane::Target> current_version;
403  if (!storage->loadInstalledVersions(ecu_serial.ToString(), &current_version, nullptr)) {
404  LOG_WARNING << "Could not load currently installed version for ECU ID: " << ecu_serial.ToString();
405  break;
406  }
407 
408  if (!current_version) {
409  LOG_WARNING << "Current version for ECU ID: " << ecu_serial.ToString() << " is unknown";
410  is_new = true;
411  } else if (current_version->filename() != target.filename()) {
412  is_new = true;
413  }
414 
415  if (primary_ecu_serial == ecu_serial) {
416  if (!target.IsOstree() &&
417  (config.pacman.type == PackageManager::kOstree || config.pacman.type == PackageManager::kOstreeDockerApp)) {
418  LOG_ERROR << "Cannot install a non-OSTree package on an OSTree system";
419  last_exception = Uptane::InvalidTarget(target.filename());
420  return false;
421  }
422  }
423 
424  if (is_new && ecus_count != nullptr) {
425  (*ecus_count)++;
426  }
427  // no updates for this image => continue
428  }
429  if (is_new) {
430  new_targets->push_back(target);
431  }
432  }
433  return true;
434 }

436 std::unique_ptr<Uptane::Target> SotaUptaneClient::findTargetHelper(const Uptane::Targets &cur_targets,
437  const Uptane::Target &queried_target,
438  const int level, const bool terminating,
439  const bool offline) {
440  TargetCompare target_comp(queried_target);
441  const auto it = std::find_if(cur_targets.targets.cbegin(), cur_targets.targets.cend(), target_comp);
442  if (it != cur_targets.targets.cend()) {
443  return std_::make_unique<Uptane::Target>(*it);
444  }
445 
446  if (terminating || level >= Uptane::kDelegationsMaxDepth) {
447  return std::unique_ptr<Uptane::Target>(nullptr);
448  }
449 
450  for (const auto &delegate_name : cur_targets.delegated_role_names_) {
451  Uptane::Role delegate_role = Uptane::Role::Delegation(delegate_name);
452  auto patterns = cur_targets.paths_for_role_.find(delegate_role);
453  if (patterns == cur_targets.paths_for_role_.end()) {
454  continue;
455  }
456 
457  bool match = false;
458  for (const auto &pattern : patterns->second) {
459  if (fnmatch(pattern.c_str(), queried_target.filename().c_str(), 0) == 0) {
460  match = true;
461  break;
462  }
463  }
464  if (!match) {
465  continue;
466  }
467 
468  // Target name matches one of the patterns
469 
470  auto delegation =
471  Uptane::getTrustedDelegation(delegate_role, cur_targets, images_repo, *storage, *uptane_fetcher, offline);
472  if (delegation.isExpired(TimeStamp::Now())) {
473  continue;
474  }
475 
476  auto is_terminating = cur_targets.terminating_role_.find(delegate_role);
477  if (is_terminating == cur_targets.terminating_role_.end()) {
478  throw Uptane::Exception("images", "Inconsistent delegations");
479  }
480 
481  auto found_target = findTargetHelper(delegation, queried_target, level + 1, is_terminating->second, offline);
482  if (found_target != nullptr) {
483  return found_target;
484  }
485  }
486 
487  return std::unique_ptr<Uptane::Target>(nullptr);
488 }

501  const api::FlowControlToken *token) {
502  // Uptane step 4 - download all the images and verify them against the metadata (for OSTree - pull without
503  // deploying)

560 std::pair<bool, Uptane::Target> SotaUptaneClient::downloadImage(const Uptane::Target &target,
561  const api::FlowControlToken *token) {
562  // TODO: support downloading encrypted targets from director
563 
564  const std::string &correlation_id = director_repo.getCorrelationId();
565  // send an event for all ecus that are touched by this target
566  for (const auto &ecu : target.ecus()) {
567  report_queue->enqueue(std_::make_unique<EcuDownloadStartedReport>(ecu.first, correlation_id));
568  }
569 
570  KeyManager keys(storage, config.keymanagerConfig());
571  keys.loadKeys();
572  auto prog_cb = [this](const Uptane::Target &t, const std::string description, unsigned int progress) {
573  report_progress_cb(events_channel.get(), t, description, progress);
574  };
575 
576  bool success = false;
577  const int max_tries = 3;
578  int tries = 0;
579  std::chrono::milliseconds wait(500);
580 
581  for (; tries < max_tries; tries++) {
582  success = package_manager_->fetchTarget(target, *uptane_fetcher, keys, prog_cb, token);
583  if (success) {
584  break;
585  } else if (tries < max_tries - 1) {
586  std::this_thread::sleep_for(wait);
587  wait *= 2;
588  }
589  }
590  if (!success) {
591  LOG_ERROR << "Download unsuccessful after " << tries << " attempts.";
592  }
593 
594  // send this asynchronously before `sendEvent`, so that the report timestamp
595  // would not be delayed by callbacks on events

704 result::UpdateCheck SotaUptaneClient::checkUpdates() {
705  result::UpdateCheck result;
706 
707  std::vector<Uptane::Target> updates;
708  unsigned int ecus_count = 0;
709  if (!uptaneIteration(&updates, &ecus_count)) {
710  result = result::UpdateCheck({}, 0, result::UpdateStatus::kError, Json::nullValue, "Could not update metadata.");
711  return result;
712  }
713 
714  std::string director_targets;
715  storage->loadNonRoot(&director_targets, Uptane::RepositoryType::Director(), Uptane::Role::Targets());
716 
717  if (updates.empty()) {
718  LOG_DEBUG << "No new updates found in Uptane metadata.";
719  result = result::UpdateCheck(updates, ecus_count, result::UpdateStatus::kNoUpdatesAvailable,
720  Utils::parseJSON(director_targets), "");
721  return result;
722  }
723 
724  // For every target in the Director Targets metadata, walk the delegation
725  // tree (if necessary) and find a matching target in the Images repo
726  // metadata.
727  for (auto &target : updates) {
728  auto images_target = findTargetInDelegationTree(target, false);
729  if (images_target == nullptr) {
730  // TODO: Could also be a missing target or delegation expiration.
731  last_exception = Uptane::TargetMismatch(target.filename());
732  LOG_ERROR << "No matching target in images targets metadata for " << target;
733  result = result::UpdateCheck(updates, ecus_count, result::UpdateStatus::kError,
734  Utils::parseJSON(director_targets), "Target mismatch.");
735  return result;
736  }
737  // If the URL from the Director is unset, but the URL from the Images repo
738  // is set, use that.
739  if (target.uri().empty() && !images_target->uri().empty()) {
740  target.setUri(images_target->uri());
741  }
742  }
743 
744  result = result::UpdateCheck(updates, ecus_count, result::UpdateStatus::kUpdatesAvailable,
745  Utils::parseJSON(director_targets), "");
746  if (updates.size() == 1) {
747  LOG_INFO << "1 new update found in Uptane metadata.";
748  } else {
749  LOG_INFO << updates.size() << " new updates found in Uptane metadata.";
750  }
751  return result;

754 result::UpdateStatus SotaUptaneClient::checkUpdatesOffline(const std::vector<Uptane::Target> &targets) {
755  if (hasPendingUpdates()) {
756  // no need in update checking if there are some pending updates
757  LOG_INFO << "An update is pending. Skipping stored metadata check until installation is complete.";
758  return result::UpdateStatus::kError;
759  }
760 
761  if (targets.empty()) {
762  LOG_WARNING << "Requested targets vector is empty. Nothing to do.";
763  return result::UpdateStatus::kError;
764  }
765 
766  std::vector<Uptane::Target> director_targets;
767  unsigned int ecus_count = 0;
768  if (!uptaneOfflineIteration(&director_targets, &ecus_count)) {
769  LOG_ERROR << "Invalid Uptane metadata in storage.";
770  return result::UpdateStatus::kError;
771  }
772 
773  if (director_targets.empty()) {
774  LOG_ERROR << "No new updates found in Uptane metadata, but expected " << targets.size() << ".";
775  return result::UpdateStatus::kNoUpdatesAvailable;
776  }
777 
778  // For every target in the Director Targets metadata, walk the delegation
779  // tree (if necessary) and find a matching target in the Images repo
780  // metadata.
781  for (const auto &target : targets) {
782  TargetCompare target_comp(target);
783  const auto it = std::find_if(director_targets.cbegin(), director_targets.cend(), target_comp);
784  if (it == director_targets.cend()) {
785  LOG_ERROR << "No matching target in director targets metadata for " << target;
786  return result::UpdateStatus::kError;
787  }
788 
789  const auto images_target = findTargetInDelegationTree(target, true);
790  if (images_target == nullptr) {
791  LOG_ERROR << "No matching target in images targets metadata for " << target;
792  return result::UpdateStatus::kError;
793  }
794  }
795 
796  return result::UpdateStatus::kUpdatesAvailable;
797 }

799 result::Install SotaUptaneClient::uptaneInstall(const std::vector<Uptane::Target> &updates) {
800  result::Install result;
801  const std::string &correlation_id = director_repo.getCorrelationId();
802 
803  // clear all old results first
804  storage->clearInstallationResults();
805 
806  // Recheck the Uptane metadata and make sure the requested updates are
807  // consistent with the stored metadata.
808  result::UpdateStatus update_status = checkUpdatesOffline(updates);
809  if (update_status != result::UpdateStatus::kUpdatesAvailable) {
810  if (update_status == result::UpdateStatus::kNoUpdatesAvailable) {
811  result.dev_report = {false, data::ResultCode::Numeric::kAlreadyProcessed, ""};
812  } else {
813  result.dev_report = {false, data::ResultCode::Numeric::kInternalError, ""};
814  }
815  storage->storeDeviceInstallationResult(result.dev_report, "Stored Uptane metadata is invalid", correlation_id);
816  sendEvent<event::AllInstallsComplete>(result);
817  return result;
818  }
819 
820  // Recheck the downloaded update hashes.
821  for (const auto &update : updates) {
822  if (package_manager_->verifyTarget(update) != TargetStatus::kGood) {
823  result.dev_report = {false, data::ResultCode::Numeric::kInternalError, ""};
824  storage->storeDeviceInstallationResult(result.dev_report, "Downloaded target is invalid", correlation_id);
825  sendEvent<event::AllInstallsComplete>(result);
826  return result;
827  }
828  }
829 
830  // Uptane step 5 (send time to all ECUs) is not implemented yet.
831  Uptane::EcuSerial primary_ecu_serial = uptane_manifest.getPrimaryEcuSerial();
832  std::vector<Uptane::Target> primary_updates = findForEcu(updates, primary_ecu_serial);
833  // 6 - send metadata to all the ECUs
834  sendMetadataToEcus(updates);
835 
836  // 7 - send images to ECUs (deploy for OSTree)
837  if (primary_updates.size() != 0u) {
838  // assuming one OSTree OS per primary => there can be only one OSTree update
839  Uptane::Target primary_update = primary_updates[0];
840  primary_update.setCorrelationId(correlation_id);
841 
842  report_queue->enqueue(std_::make_unique<EcuInstallationStartedReport>(primary_ecu_serial, correlation_id));
843  sendEvent<event::InstallStarted>(primary_ecu_serial);
844 
845  data::InstallationResult install_res;
846  if (!isInstalledOnPrimary(primary_update)) {
847  // notify the bootloader before installation happens, because installation is not atomic and
848  // a false notification doesn't hurt when rollbacks are implemented
849  package_manager_->updateNotify();
850  install_res = PackageInstallSetResult(primary_update);
851  if (install_res.result_code.num_code == data::ResultCode::Numeric::kNeedCompletion) {
852  // update needs a reboot, send distinct EcuInstallationApplied event
853  report_queue->enqueue(std_::make_unique<EcuInstallationAppliedReport>(primary_ecu_serial, correlation_id));
854  sendEvent<event::InstallTargetComplete>(primary_ecu_serial, true); // TODO: distinguish from success here?
855  } else if (install_res.result_code.num_code == data::ResultCode::Numeric::kOk) {
856  storage->saveEcuInstallationResult(primary_ecu_serial, install_res);
857  report_queue->enqueue(
858  std_::make_unique<EcuInstallationCompletedReport>(primary_ecu_serial, correlation_id, true));
859  sendEvent<event::InstallTargetComplete>(primary_ecu_serial, true);
860  } else {
861  // general error case
862  storage->saveEcuInstallationResult(primary_ecu_serial, install_res);
863  report_queue->enqueue(
864  std_::make_unique<EcuInstallationCompletedReport>(primary_ecu_serial, correlation_id, false));
865  sendEvent<event::InstallTargetComplete>(primary_ecu_serial, false);
866  }
867  } else {
868  install_res = data::InstallationResult(data::ResultCode::Numeric::kAlreadyProcessed, "Package already installed");
869  storage->saveEcuInstallationResult(primary_ecu_serial, install_res);
870  // TODO: distinguish this case from regular failure for local and remote
871  // event reporting
872  report_queue->enqueue(std_::make_unique<EcuInstallationCompletedReport>(uptane_manifest.getPrimaryEcuSerial(),
873  correlation_id, false));
874  sendEvent<event::InstallTargetComplete>(uptane_manifest.getPrimaryEcuSerial(), false);
875  }
876  result.ecu_reports.emplace(result.ecu_reports.begin(), primary_update, uptane_manifest.getPrimaryEcuSerial(),
877  install_res);
878  // TODO: other updates for primary
879  } else {
880  LOG_INFO << "No update to install on primary";
881  }
882 
883  auto sec_reports = sendImagesToEcus(updates);
884  result.ecu_reports.insert(result.ecu_reports.end(), sec_reports.begin(), sec_reports.end());
885  computeDeviceInstallationResult(&result.dev_report, correlation_id);
886  sendEvent<event::AllInstallsComplete>(result);
887 
888  if (!(result.dev_report.isSuccess() || result.dev_report.needCompletion())) {
889  director_repo.dropTargets(*storage); // fix for OTA-2587, listen to backend again after end of install
890  }
891 
892  return result;
893 }

965 // Check stored secondaries list against secondaries known to aktualizr.
966 void SotaUptaneClient::verifySecondaries() {
967  storage->clearMisconfiguredEcus();
968  EcuSerials serials;
969  if (!storage->loadEcuSerials(&serials) || serials.empty()) {
970  LOG_ERROR << "No ECU serials found in storage!";
971  return;
972  }
973 
974  std::vector<MisconfiguredEcu> misconfigured_ecus;
975  std::vector<bool> found(serials.size(), false);
976  SerialCompare primary_comp(uptane_manifest.getPrimaryEcuSerial());
977  EcuSerials::const_iterator store_it;
978  store_it = std::find_if(serials.cbegin(), serials.cend(), primary_comp);
979  if (store_it == serials.cend()) {
980  LOG_ERROR << "Primary ECU serial " << uptane_manifest.getPrimaryEcuSerial() << " not found in storage!";
981  misconfigured_ecus.emplace_back(uptane_manifest.getPrimaryEcuSerial(), Uptane::HardwareIdentifier(""),
982  EcuState::kOld);
983  } else {
984  found[static_cast<size_t>(std::distance(serials.cbegin(), store_it))] = true;
985  }
986 
987  std::map<Uptane::EcuSerial, std::shared_ptr<Uptane::SecondaryInterface>>::const_iterator it;
988  for (it = secondaries.cbegin(); it != secondaries.cend(); ++it) {
989  SerialCompare secondary_comp(it->second->getSerial());
990  store_it = std::find_if(serials.cbegin(), serials.cend(), secondary_comp);
991  if (store_it == serials.cend()) {
992  LOG_ERROR << "Secondary ECU serial " << it->second->getSerial() << " (hardware ID " << it->second->getHwId()
993  << ") not found in storage!";
994  misconfigured_ecus.emplace_back(it->second->getSerial(), it->second->getHwId(), EcuState::kNotRegistered);
995  } else if (found[static_cast<size_t>(std::distance(serials.cbegin(), store_it))]) {
996  LOG_ERROR << "Secondary ECU serial " << it->second->getSerial() << " (hardware ID " << it->second->getHwId()
997  << ") has a duplicate entry in storage!";
998  } else {
999  found[static_cast<size_t>(std::distance(serials.cbegin(), store_it))] = true;
1000  }
1001  }
1002 
1003  std::vector<bool>::iterator found_it;
1004  for (found_it = found.begin(); found_it != found.end(); ++found_it) {
1005  if (!*found_it) {
1006  auto not_registered = serials[static_cast<size_t>(std::distance(found.begin(), found_it))];
1007  LOG_WARNING << "ECU serial " << not_registered.first << " in storage was not reported to aktualizr!";
1008  misconfigured_ecus.emplace_back(not_registered.first, not_registered.second, EcuState::kOld);
1009  }
1010  }
1011 
1012  storage->storeMisconfiguredEcus(misconfigured_ecus);
1013 }

1120  continue;
1121  }

1128  }

1138  sendFirmwareAsync(sec, std::make_shared<std::string>(creds_archive)));
