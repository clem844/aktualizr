218 std::string Utils::extractField(const std::string &in, unsigned int field_id) {
219  std::string out;
220  auto it = in.begin();
221 
222  // skip spaces
223  for (; it != in.end() && (isspace(*it) != 0); it++) {
224  ;
225  }
226  for (unsigned int k = 0; k < field_id; k++) {
227  bool empty = true;
228  for (; it != in.end() && (isspace(*it) == 0); it++) {
229  empty = false;
230  }
231  if (empty) {
232  throw std::runtime_error(std::string("No such field ").append(std::to_string(field_id)));
233  }
234  for (; it != in.end() && (isspace(*it) != 0); it++) {
235  ;
236  }
237  }
238 
239  for (; it != in.end() && (isspace(*it) == 0); it++) {
240  out += *it;
241  }
242  return out;
243 }

300 static ssize_t read_cb(struct archive *a, void *client_data, const void **buffer) {
301  auto s = reinterpret_cast<archive_state *>(client_data);
302  if (s->is.fail()) {
303  archive_set_error(a, -1, "unable to read from stream");
304  return 0;
305  }
306  if (s->is.eof()) {
307  return 0;
308  }
309  s->is.read(s->buf.data(), BSIZE);
310  if (!s->is.eof() && s->is.fail()) {
311  archive_set_error(a, -1, "unable to read from stream");
312  return 0;
313  }
314  *buffer = s->buf.data();
315 
316  return s->is.gcount();
317 }

366 Json::Value Utils::getNetworkInfo() {
367  // get interface with default route
368  std::ifstream path_stream("/proc/net/route");
369  std::string route_content((std::istreambuf_iterator<char>(path_stream)), std::istreambuf_iterator<char>());
370 
371  struct Itf {
372  std::string name = std::string();
373  std::string ip = std::string();
374  std::string mac = std::string();
375  } itf;
376  std::istringstream route_stream(route_content);
377  std::array<char, 200> line{};
378 
379  // skip first line
380  route_stream.getline(&line[0], line.size());
381  while (route_stream.getline(&line[0], line.size())) {
382  std::string itfn = Utils::extractField(&line[0], 0);
383  std::string droute = Utils::extractField(&line[0], 1);
384  if (droute == "00000000") {
385  itf.name = itfn;
386  // take the first routing to 0
387  break;
388  }
389  }
390 
391  if (itf.name != "") {
392  {
393  // get ip address
394  StructGuard<struct ifaddrs> ifaddrs(nullptr, freeifaddrs);
395  {
396  struct ifaddrs *ifa;
397  if (getifaddrs(&ifa) < 0) {
398  LOG_ERROR << "getifaddrs: " << std::strerror(errno);
399  } else {
400  ifaddrs.reset(ifa);
401  }
402  }
403  if (ifaddrs != nullptr) {
404  for (struct ifaddrs *ifa = ifaddrs.get(); ifa != nullptr; ifa = ifa->ifa_next) {
405  if (itf.name == ifa->ifa_name) {
406  if (ifa->ifa_addr == nullptr) {
407  continue;
408  }
409  if (ifa->ifa_addr->sa_family != AF_INET) {
410  continue;
411  }
412  const struct sockaddr_storage *sa = reinterpret_cast<struct sockaddr_storage *>(ifa->ifa_addr);
413 
414  itf.ip = Utils::ipDisplayName(*sa);
415  }
416  }
417  }
418  }
419  {
420  // get mac address
421  std::ifstream mac_stream("/sys/class/net/" + itf.name + "/address");
422  std::string m((std::istreambuf_iterator<char>(mac_stream)), std::istreambuf_iterator<char>());
423  itf.mac = std::move(m);
424  boost::trim_right(itf.mac);
425  }
426  }
427 
428  Json::Value network_info;
429  network_info["local_ipv4"] = itf.ip;
430  network_info["mac"] = itf.mac;
431  network_info["hostname"] = Utils::getHostname();
432 
433  return network_info;
434 }

466 std::string Utils::readFileFromArchive(std::istream &as, const std::string &filename, const bool trim) {
467  struct archive *a = archive_read_new();
468  if (a == nullptr) {
469  LOG_ERROR << "archive error: could not initialize archive object";
470  throw std::runtime_error("archive error");
471  }
472  archive_read_support_filter_all(a);
473  archive_read_support_format_all(a);
474  auto state = std_::make_unique<archive_state>(std::ref(as));
475  int r = archive_read_open(a, reinterpret_cast<void *>(state.get()), nullptr, read_cb, nullptr);
476  if (r != ARCHIVE_OK) {
477  LOG_ERROR << "archive error: " << archive_error_string(a);
478  archive_read_free(a);
479  throw std::runtime_error("archive error");
480  }
481 
482  bool found = false;
483  std::stringstream out_stream;
484  struct archive_entry *entry;
485  while (archive_read_next_header(a, &entry) == ARCHIVE_OK) {
486  if (filename != archive_entry_pathname(entry)) {
487  archive_read_data_skip(a);
488  continue;
489  }
490 
491  const char *buff;
492  size_t size;
493  int64_t offset;
494 
495  for (;;) {
496  r = archive_read_data_block(a, reinterpret_cast<const void **>(&buff), &size, &offset);
497  if (r == ARCHIVE_EOF) {
498  found = true;
499  break;
500  }
501  if (r != ARCHIVE_OK) {
502  LOG_ERROR << "archive error: " << archive_error_string(a);
503  break;
504  }
505  if (size > 0 && buff != nullptr) {
506  out_stream.write(buff, static_cast<ssize_t>(size));
507  }
508  }
509  }
510 
511  r = archive_read_free(a);
512  if (r != ARCHIVE_OK) {
513  LOG_ERROR << "archive error: " << archive_error_string(a);
514  }
515 
516  if (!found) {
517  throw std::runtime_error("could not extract " + filename + " from archive");
518  }
519 
520  std::string result = out_stream.str();
521  if (trim) {
522  boost::trim_if(result, boost::is_any_of(" \t\r\n"));
523  }
524  return result;
525 }

538 void Utils::writeArchive(const std::map<std::string, std::string> &entries, std::ostream &as) {
539  struct archive *a = archive_write_new();
540  if (a == nullptr) {
541  LOG_ERROR << "archive error: could not initialize archive object";
542  throw std::runtime_error("archive error");
543  }
544  archive_write_set_format_pax(a);
545  archive_write_add_filter_gzip(a);
546 
547  int r = archive_write_open(a, reinterpret_cast<void *>(&as), nullptr, write_cb, nullptr);
548  if (r != ARCHIVE_OK) {
549  LOG_ERROR << "archive error: " << archive_error_string(a);
550  archive_write_free(a);
551  throw std::runtime_error("archive error");
552  }
553 
554  struct archive_entry *entry = archive_entry_new();
555  for (const auto &el : entries) {
556  archive_entry_clear(entry);
557  archive_entry_set_filetype(entry, AE_IFREG);
558  archive_entry_set_perm(entry, S_IRWXU | S_IRWXG | S_IRWXO);
559  archive_entry_set_size(entry, static_cast<ssize_t>(el.second.size()));
560  archive_entry_set_pathname(entry, el.first.c_str());
561  if (archive_write_header(a, entry) != 0) {
562  LOG_ERROR << "archive error: " << archive_error_string(a);
563  archive_entry_free(entry);
564  archive_write_free(a);
565  throw std::runtime_error("archive error");
566  }
567  if (archive_write_data(a, el.second.c_str(), el.second.size()) < 0) {
568  LOG_ERROR << "archive error: " << archive_error_string(a);
569  archive_entry_free(entry);
570  archive_write_free(a);
571  throw std::runtime_error("archive error");
572  }
573  }
574  archive_entry_free(entry);
575  r = archive_write_free(a);
576  if (r != ARCHIVE_OK) {
577  LOG_ERROR << "archive error: " << archive_error_string(a);
578  }
579 }

592  char ipstr[INET6_ADDRSTRLEN];
593 
594  switch (saddr.ss_family) {

603  return std::string(ipstr);
604  }
605  default:
606  return "unknown";
607  }

701  c == '_' || c == '~' || c == '/') {
702  res.push_back(c);
