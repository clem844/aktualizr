88 bool AktualizrSecondary::sendFirmwareResp(const std::shared_ptr<std::string>& firmware) {
89  auto targetsForThisEcu = director_repo_.getTargets(getSerial(), getHardwareID());
90 
91  if (targetsForThisEcu.size() != 1) {
92  LOG_ERROR << "Invalid number of targets (should be one): " << targetsForThisEcu.size();
93  return false;
94  }
95  auto target_to_apply = targetsForThisEcu[0];
96 
97  std::string treehub_server;
98  std::size_t firmware_size = firmware->length();
99 
100  if (target_to_apply.IsOstree()) {
101  // this is the ostree specific case
102  try {
103  std::string ca, cert, pkey, server_url;
104  extractCredentialsArchive(*firmware, &ca, &cert, &pkey, &server_url);
105  keys_.loadKeys(&ca, &cert, &pkey);
106  boost::trim(server_url);
107  treehub_server = server_url;
108  firmware_size = 0;
109  } catch (std::runtime_error& exc) {
110  LOG_ERROR << exc.what();
111 
112  return false;
113  }
114  }
115 
116  data::InstallationResult install_res;
117 
118  if (target_to_apply.IsOstree()) {
119 #ifdef BUILD_OSTREE
120  install_res = OstreeManager::pull(config_.pacman.sysroot, treehub_server, keys_, target_to_apply);
121 
122  if (install_res.result_code.num_code != data::ResultCode::Numeric::kOk) {
123  LOG_ERROR << "Could not pull from OSTree (" << install_res.result_code.toString()
124  << "): " << install_res.description;
125  return false;
126  }
127 #else
128  LOG_ERROR << "Could not pull from OSTree. Aktualizr was built without OSTree support!";
129  return false;
130 #endif
131  } else if (pacman->name() == "debian") {
132  // TODO save debian package here.
133  LOG_ERROR << "Installation of debian images is not suppotrted yet.";
134  return false;
135  }
136 
137  if (target_to_apply.length() != firmware_size) {
138  LOG_ERROR << "The target image size specified in metadata " << target_to_apply.length()
139  << " does not match actual size " << firmware->length();
140  return false;
141  }
142 
143  if (!target_to_apply.IsOstree()) {
144  auto target_hashes = target_to_apply.hashes();
145  if (target_hashes.size() == 0) {
146  LOG_ERROR << "No hash found in the target metadata: " << target_to_apply.filename();
147  return false;
148  }
149 
150  try {
151  auto received_image_data_hash = Uptane::Hash::generate(target_hashes[0].type(), *firmware);
152 
153  if (!target_to_apply.MatchHash(received_image_data_hash)) {
154  LOG_ERROR << "The received image data hash doesn't match the hash specified in the target metadata,"
155  " hash type: "
156  << target_hashes[0].TypeString();
157  return false;
158  }
159 
160  } catch (const std::exception& exc) {
161  LOG_ERROR << "Failed to generate a hash of the received image data: " << exc.what();
162  return false;
163  }
164  }
165 
166  install_res = pacman->install(target_to_apply);
167  if (install_res.result_code.num_code != data::ResultCode::Numeric::kOk &&
168  install_res.result_code.num_code != data::ResultCode::Numeric::kNeedCompletion) {
169  LOG_ERROR << "Could not install target (" << install_res.result_code.toString() << "): " << install_res.description;
170  return false;
171  }
172 
173  if (install_res.result_code.num_code == data::ResultCode::Numeric::kNeedCompletion) {
174  storage_->saveInstalledVersion(getSerialResp().ToString(), target_to_apply, InstalledVersionUpdateMode::kPending);
175  } else {
176  storage_->saveInstalledVersion(getSerialResp().ToString(), target_to_apply, InstalledVersionUpdateMode::kCurrent);
177  }
178 
179  // TODO: https://saeljira.it.here.com/browse/OTA-4174
180  // - return data::InstallationResult to Primary
181  // - add finaliztion support, pacman->finalizeInstall() must be called at secondary startup if there is pending
182  // version
183 
184  return true;
185 }

223 bool AktualizrSecondary::doFullVerification(const Metadata& metadata) {
224  // 5.4.4.2. Full verification https://uptane.github.io/uptane-standard/uptane-standard.html#metadata_verification
225 
226  // 1. Load and verify the current time or the most recent securely attested time.
227  //
228  // We trust the time that the given system/OS/ECU provides, In ECU We Trust :)
229  TimeStamp now(TimeStamp::Now());
230 
231  // 2. Download and check the Root metadata file from the Director repository, following the procedure in
232  // Section 5.4.4.3. DirectorRepository::updateMeta() method implements this verification step, certain steps are
233  // missing though. see the method source code for details
234 
235  // 3. NOT SUPPORTED: Download and check the Timestamp metadata file from the Director repository, following the
236  // procedure in Section 5.4.4.4.
237  // 4. NOT SUPPORTED: Download and check the Snapshot metadata file from the Director repository, following the
238  // procedure in Section 5.4.4.5.
239  //
240  // 5. Download and check the Targets metadata file from the Director repository, following the procedure in
241  // Section 5.4.4.6. DirectorRepository::updateMeta() method implements this verification step
242  //
243  // The followin steps of the Director's target metadata verification are missing in DirectorRepository::updateMeta()
244  // 6. If checking Targets metadata from the Director repository, verify that there are no delegations.
245  // 7. If checking Targets metadata from the Director repository, check that no ECU identifier is represented more
246  // than once.
247  if (!director_repo_.updateMeta(*storage_, metadata)) {
248  LOG_ERROR << "Failed to update director metadata: " << director_repo_.getLastException().what();
249  return false;
250  }
251 
252  auto targetsForThisEcu = director_repo_.getTargets(getSerial(), getHardwareID());
253 
254  if (targetsForThisEcu.size() != 1) {
255  LOG_ERROR << "Invalid number of targets (should be 1): " << targetsForThisEcu.size();
256  return false;
257  }
258 
259  // 6. Download and check the Root metadata file from the Image repository, following the procedure in Section 5.4.4.3.
260  // 7. Download and check the Timestamp metadata file from the Image repository, following the procedure in
261  // Section 5.4.4.4.
262  // 8. Download and check the Snapshot metadata file from the Image repository, following the procedure in
263  // Section 5.4.4.5.
264  // 9. Download and check the top-level Targets metadata file from the Image repository, following the procedure in
265  // Section 5.4.4.6.
266  if (!image_repo_.updateMeta(*storage_, metadata)) {
267  LOG_ERROR << "Failed to update image metadata: " << image_repo_.getLastException().what();
268  return false;
269  }
270 
271  // 10. Verify that Targets metadata from the Director and Image repositories match.
272  if (!(director_repo_.getTargets() == *image_repo_.getTargets())) {
273  LOG_ERROR << "Targets metadata from the Director and Image repositories DOES NOT match ";
274  return false;
275  }
276 
277  return true;
278 }
